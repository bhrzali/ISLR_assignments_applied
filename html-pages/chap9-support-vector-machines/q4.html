class = sample(100,50)
x = rnorm(200)
x2 = 2*x^2 + 3*x + 4 + rnorm(200)
x2[class]=x2[class]+3
x2[-class]=x2[-class]-3
y=rep(0,200)
y[class]=1
y[-class]=2
dat = data.frame(x=cbind(x,x2),y=as.factor(y))
